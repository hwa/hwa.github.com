cd project
git init

vi .git/info/exclude

#edit files


git add .
git commit -a -m "comments is required"

#edit again

git diff # show uncommited diff
git commit -a -m 'commit another new'

git branch new-branch
git branch # list all branches
git checkout new-branch

#edit files

git commit -a -m 'comment into new branch'

git checkout master #master is default branch

git merge new-branch
git branch -d new-branch  #delete new branch

git checkout ac9f9a # jump to a historical commit
git checkout master # go back to master HEAD, newest commit

git revert ac9f9a   # make a new commit that is identical to this historical commit
